---
title: "Copy Positions"
api: "POST https://api.copin.io/copy-positions/page"
description: "Retrieve copy positions with pagination"
---

## Authentication
This endpoint requires authentication using a JWT token in the Authorization header.

<ParamField header="Authorization" type="string" required>
    JWT token in the format: `Authorization {token}`
</ParamField>

## Request Query

<ParamField query="limit" type="number" default="20">
    Number of records to return per page
</ParamField>
<ParamField query="offset" type="number" default="0">
    Number of records to skip for pagination
</ParamField>
<ParamField query="status" type="string">
    Current status of the position ("OPEN/CLOSE")
</ParamField>
<ParamField query="isLong" type="boolean">
    Position side
</ParamField>

## Response
The response includes a data array containing position objects and a meta object with pagination information.

<ResponseField name="data" type="array">
    Array of copy position objects
    <Expandable title="properties">
        <ResponseField name="id" type="string">
            Unique identifier for the copy position
        </ResponseField>
        <ResponseField name="userId" type="string">
            Identifier of the user who owns this position
        </ResponseField>
        <ResponseField name="copyTradeId" type="string">
            Identifier of the copy trade configuration that created this position
        </ResponseField>
        <ResponseField name="copyAccount" type="string">
            The wallet address being copied from
        </ResponseField>
        <ResponseField name="positionIndex" type="number">
            Index of the position in the original account
        </ResponseField>
        <ResponseField name="copyPositionIndex" type="number">
            Index of the position in the copy account
        </ResponseField>
        <ResponseField name="indexToken" type="string">
            Token identifier for the indexed token
        </ResponseField>
        <ResponseField name="identifyKey" type="string">
            Unique key to identify the position
        </ResponseField>
        <ResponseField name="pair" type="string">
            Trading pair (e.g., "ETH-USDT")
        </ResponseField>
        <ResponseField name="key" type="string">
            Unique key for the position
        </ResponseField>
        <ResponseField name="sourceOrderTxHashes" type="array">
            Array of transaction hashes for the source orders
        </ResponseField>
        <ResponseField name="sourceSizeDelta" type="string">
            Size delta in the source position
        </ResponseField>
        <ResponseField name="sizeDelta" type="string">
            Size delta in the copy position
        </ResponseField>
        <ResponseField name="totalSizeDelta" type="number">
            Total size delta for the position
        </ResponseField>
        <ResponseField name="sourceInitialVol" type="string">
            Initial volume in the source position
        </ResponseField>
        <ResponseField name="initialVol" type="number">
            Initial volume in the copy position
        </ResponseField>
        <ResponseField name="isLong" type="boolean">
            Whether the position is long (true) or short (false)
        </ResponseField>
        <ResponseField name="entryPrice" type="number">
            Entry price for the position
        </ResponseField>
        <ResponseField name="leverage" type="number">
            Leverage used for the position
        </ResponseField>
        <ResponseField name="stopLossPrice" type="number">
            Price at which stop loss is triggered
        </ResponseField>
        <ResponseField name="stopLossAmount" type="number">
            Amount for stop loss
        </ResponseField>
        <ResponseField name="takeProfitPrice" type="number">
            Price at which take profit is triggered
        </ResponseField>
        <ResponseField name="takeProfitAmount" type="number">
            Amount for take profit
        </ResponseField>
        <ResponseField name="pnl" type="number">
            Total profit and loss
        </ResponseField>
        <ResponseField name="realisedPnl" type="number">
            Realised profit and loss
        </ResponseField>
        <ResponseField name="funding" type="number">
            Funding payments for the position
        </ResponseField>
        <ResponseField name="fee" type="number">
            Fees paid for the position
        </ResponseField>
        <ResponseField name="closeType" type="string">
            How the position was closed (COPY_TRADE/MANUAL/FORCE_CLOSE/TAKE_PROFIT/STOP_LOSS/LIQUIDATE/OVERWRITE)
        </ResponseField>
        <ResponseField name="status" type="string">
            Current status of the position (OPEN/CLOSE)
        </ResponseField>
        <ResponseField name="protocol" type="string">
            Protocol used for the position
        </ResponseField>
        <ResponseField name="exchange" type="string">
            Exchange where the position was opened
        </ResponseField>
        <ResponseField name="isReverse" type="boolean">
            Whether this is a reverse copy position
        </ResponseField>
        <ResponseField name="lastOrderAt" type="string">
            Timestamp of the last order for this position
        </ResponseField>
        <ResponseField name="createdAt" type="string">
            Timestamp when the position was created
        </ResponseField>
        <ResponseField name="copyTradeTitle" type="string">
            Title of the copy trade configuration
        </ResponseField>
    </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
    Metadata about the response
    <Expandable title="properties">
        <ResponseField name="limit" type="number">
            Number of records returned
        </ResponseField>
        <ResponseField name="offset" type="number">
            Number of records skipped
        </ResponseField>
        <ResponseField name="total" type="number">
            Total number of records matching the query
        </ResponseField>
        <ResponseField name="totalPages" type="number">
            Total number of pages
        </ResponseField>
    </Expandable>
</ResponseField>
