---
title: "Activity Logs"
api: "POST https://api.copin.io/activity-logs/page"
description: "Retrieve activity logs with pagination"
---

## Authentication
This endpoint requires authentication using a JWT token in the Authorization header.

<ParamField header="Authorization" type="string" required>
    JWT token in the format: `Authorization {token}`
</ParamField>

## Request Body

<ParamField body="pagination" type="object">
    Pagination parameters
    <Expandable title="properties">
        <ParamField body="limit" type="number" default="20">
            Number of records to return
        </ParamField>
        <ParamField body="offset" type="number" default="0">
            Number of records to skip
        </ParamField>
    </Expandable>
</ParamField>

<ParamField body="sortBy" type="string" default="createdAt">
    Field to sort the results by
</ParamField>

<ParamField body="sortType" type="string" default="desc">
    Sort direction, either "asc" or "desc"
</ParamField>

## Response
The response includes a data array containing activity log objects and a meta object with pagination information.

<ResponseField name="data" type="array">
    Array of activity log objects
    <Expandable title="properties">
        <ResponseField name="id" type="string">
            Unique identifier for the activity log
        </ResponseField>
        <ResponseField name="protocol" type="string">
            Protocol used for the activity
        </ResponseField>
        <ResponseField name="sourceAccount" type="string">
            The source account address that was copied from
        </ResponseField>
        <ResponseField name="userId" type="string">
            Identifier of the user who owns this activity
        </ResponseField>
        <ResponseField name="copyOrderId" type="string">
            Identifier of the copy order
        </ResponseField>
        <ResponseField name="copyTradeTitle" type="string">
            Display name of the copy trade configuration
        </ResponseField>
        <ResponseField name="exchange" type="string">
            Exchange where the activity occurred
        </ResponseField>
        <ResponseField name="indexToken" type="string">
            Token identifier for the indexed token
        </ResponseField>
        <ResponseField name="pair" type="string">
            Trading pair (e.g., "ETH-USDT")
        </ResponseField>
        <ResponseField name="price" type="number">
            Execution price for the activity
        </ResponseField>
        <ResponseField name="sourcePrice" type="number">
            Price in the source transaction
        </ResponseField>
        <ResponseField name="copyTradeId" type="string">
            Identifier of the copy trade configuration
        </ResponseField>
        <ResponseField name="volume" type="number">
            Volume amount for the activity
        </ResponseField>
        <ResponseField name="leverage" type="number">
            Leverage used for the activity
        </ResponseField>
        <ResponseField name="isLong" type="boolean">
            Whether the position is long (true) or short (false)
        </ResponseField>
        <ResponseField name="type" type="string">
            Type of activity (e.g., "OPEN", "CLOSE")
        </ResponseField>
        <ResponseField name="isSuccess" type="boolean">
            Whether the activity was successful
        </ResponseField>
        <ResponseField name="sourceTxHash" type="string">
            Transaction hash from the source
        </ResponseField>
        <ResponseField name="copyWalletId" type="string">
            Identifier of the copy wallet
        </ResponseField>
        <ResponseField name="copyWalletName" type="string">
            Name of the copy wallet
        </ResponseField>
        <ResponseField name="cexOrderId" type="string">
            Order ID in the centralized exchange, if applicable
        </ResponseField>
        <ResponseField name="isReverse" type="boolean">
            Whether this is a reverse copy
        </ResponseField>
        <ResponseField name="copyPositionId" type="string">
            Identifier of the copy position
        </ResponseField>
        <ResponseField name="createdAt" type="string">
            Timestamp when the activity occurred
        </ResponseField>
    </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
    Metadata about the response
    <Expandable title="properties">
        <ResponseField name="limit" type="number">
            Number of records returned
        </ResponseField>
        <ResponseField name="offset" type="number">
            Number of records skipped
        </ResponseField>
        <ResponseField name="total" type="number">
            Total number of records matching the query
        </ResponseField>
        <ResponseField name="totalPages" type="number">
            Total number of pages
        </ResponseField>
    </Expandable>
</ResponseField>
