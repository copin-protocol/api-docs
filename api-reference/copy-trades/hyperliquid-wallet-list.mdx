## Authentication
This endpoint requires authentication using a JWT token in the Authorization header.

<ParamField header="Authorization" type="string" required>
    JWT token in the format: `Authorization {token}`
</ParamField>

## Response
The API returns an array of copy trade wallet objects, each containing details about a single copy trade wallet.

<ResponseField name="data" type="array">
    Array of copy wallet objects
    <Expandable title="properties">
        <ResponseField name="id" type="string">
            Unique identifier for the copy trade wallet
        </ResponseField>
        <ResponseField name="userId" type="string">
            Identifier of the user who owns this wallet
        </ResponseField>
        <ResponseField name="name" type="string">
            Display name of the copy trade wallet
        </ResponseField>
        <ResponseField name="exchange" type="string">
            The exchange platform this wallet is connected to
        </ResponseField>
        <ResponseField name="hyperliquid" type="object">
            Details specific to HyperLiquid wallets
        </ResponseField>
        <ResponseField name="hyperliquid.apiKey" type="string">
            API key for the HyperLiquid wallet
        </ResponseField>
        <ResponseField name="hyperliquid.embeddedWallet" type="string">
            Wallet address for embedded wallets
        </ResponseField>
        <ResponseField name="hyperliquid.isEmbedded" type="boolean">
            Indicates if this is an embedded wallet
        </ResponseField>
        <ResponseField name="hyperliquid.verified" type="boolean">
            Indicates if the wallet has been verified
        </ResponseField>
        <ResponseField name="balance" type="number">
            Total balance in the wallet
        </ResponseField>
        <ResponseField name="availableBalance" type="number">
            Available balance that can be used for trading
        </ResponseField>
        <ResponseField name="createdAt" type="string">
            Timestamp when the wallet was created
        </ResponseField>
        <ResponseField name="pnl24h" type="number">
            Pnl during 24h
        </ResponseField>
        <ResponseField name="totalPnl" type="number">
            Total pnl
        </ResponseField>
    </Expandable>
</ResponseField>
