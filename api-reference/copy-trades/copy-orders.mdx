## Authentication
This endpoint requires authentication using a JWT token in the Authorization header.

<ParamField header="Authorization" type="string" required>
    JWT token in the format: `Authorization {token}`
</ParamField>

## Path Parameters

<ParamField path="id" type="string" required>
    Unique identifier of the copy position to retrieve orders for
</ParamField>

## Response
The API returns an array of order objects for the specified copy position, sorted by creation date (newest first).

<ResponseField name="data" type="array">
    Array of copy order objects
    <Expandable title="properties">
        <ResponseField name="price" type="number">
            Price at which the order was executed
        </ResponseField>
        <ResponseField name="size" type="number">
            Size of the order
        </ResponseField>
        <ResponseField name="pnl" type="number">
            Profit and loss from this order
        </ResponseField>
        <ResponseField name="realisedPnl" type="number">
            Realized profit and loss from this order
        </ResponseField>
        <ResponseField name="fee" type="number">
            Fees paid for this order
        </ResponseField>
        <ResponseField name="isLong" type="boolean">
            Whether this is a long position (true) or short position (false)
        </ResponseField>
        <ResponseField name="isIncrease" type="boolean">
            Whether this order increases the position size (true) or decreases it (false)
        </ResponseField>
        <ResponseField name="protocol" type="string">
            Protocol used for this order
        </ResponseField>
        <ResponseField name="txHash" type="string">
            Transaction hash for the order
        </ResponseField>
        <ResponseField name="createdAt" type="string">
            Timestamp when the order was created
        </ResponseField>
    </Expandable>
</ResponseField>
