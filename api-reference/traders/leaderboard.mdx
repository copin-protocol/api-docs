---
title: "Trader Leaderboard"
api: "GET https://api.copin.io/leaderboards/page"
description: "Retrieve trader leaderboard data"
---

## Query Parameters

<ParamField query="protocol" type="string" default="GMX" required>
  The protocol to get leaderboard for. Supported values include: `GMX`,
  `KWENTA`, `POLYNOMIAL`, `POLYNOMIAL_L2`, `GMX_V2_AVAX`, `GMX_V2`, `GNS`,
  `GNS_POLY`, `GNS_BASE`, `GNS_APE`, `LEVEL_BNB`, `LEVEL_ARB`, `MUX_ARB`,
  `APOLLOX_BNB`, `AVANTIS_BASE`, `EQUATION_ARB`, `LOGX_BLAST`, `LOGX_MODE`,
  `MYX_ARB`, `DEXTORO`, `VELA_ARB`, `HMX_ARB`, `SYNTHETIX_V3`,
  `SYNTHETIX_V3_ARB`, `KTX_MANTLE`, `CYBERDEX`, `YFX_ARB`, `KILOEX_OPBNB`,
  `KILOEX_BNB`, `KILOEX_MANTA`, `KILOEX_BASE`, `ROLLIE_SCROLL`, `MUMMY_FANTOM`,
  `HYPERLIQUID`, `SYNFUTURE_BASE`, `MORPHEX_FANTOM`, `PERENNIAL_ARB`,
  `BSX_BASE`, `DYDX`, `UNIDEX_ARB`, `VERTEX_ARB`, `HORIZON_BNB`,
  `HOLDSTATION_ZKSYNC`, `ZENO_METIS`, `LINEHUB_LINEA`, `BMX_BASE`, `FOXIFY_ARB`,
  `APOLLOX_BASE`, `GMX_AVAX`, `SYNTHETIX`, `DEPERP_BASE`, `ELFI_ARB`
</ParamField>

<ParamField query="queryDate" type="number" default="1701835880676" required>
  Date in timestamp format
</ParamField>

<ParamField query="statisticType" type="string" default="MONTH" required>
  Type of statistics. Values: `MONTH`, `WEEK`
</ParamField>

<ParamField query="limit" type="number">
  Number of records to return (default: 20)
</ParamField>

<ParamField query="offset" type="number">
  Number of records to skip (default: 0)
</ParamField>

<ParamField query="sort_by" type="string">
  Field to sort by. Values: `ranking`, `totalPnl`, `totalRealisedPnl`,
  `totalVolume`, `totalFee`, `totalTrade`, `totalWin`, `totalLose`,
  `totalLiquidation`, `totalLiquidationAmount`
</ParamField>

<ParamField query="sort_type" type="string">
  Sort direction. Values: `asc`, `desc`
</ParamField>

## Request Example

```bash
curl -X GET "https://api.copin.io/leaderboards/page?protocol=KWENTA&queryDate=1701835880676&statisticType=MONTH&limit=20&offset=0&sort_by=ranking&sort_type=asc" \
  -H "X-COPIN_API_KEY: YOUR_COPIN_API_KEY"
```

## Response

<ResponseField name="data" type="array">
  Array of traders in the leaderboard
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Unique identifier for the leaderboard entry
    </ResponseField>
    <ResponseField name="account" type="string">
      Trader's account address
    </ResponseField>
    <ResponseField name="protocol" type="string">
      Protocol name
    </ResponseField>
    <ResponseField name="statisticType" type="string">
      Type of statistics (MONTH, WEEK)
    </ResponseField>
    <ResponseField name="rankingAt" type="string">
      Timestamp when ranking was calculated
    </ResponseField>
    <ResponseField name="ranking" type="number">
      Current ranking position
    </ResponseField>
    <ResponseField name="lastRanking" type="number">
      Previous ranking position
    </ResponseField>
    <ResponseField name="totalPnl" type="number">
      Total profit and loss including fee
    </ResponseField>
    <ResponseField name="totalRealisedPnl" type="number">
      Total profit and loss without fee
    </ResponseField>
    <ResponseField name="totalVolume" type="number">
      Total trading volume
    </ResponseField>
    <ResponseField name="totalFee" type="number">
      Total fees paid
    </ResponseField>
    <ResponseField name="totalTrade" type="number">
      Total number of trades
    </ResponseField>
    <ResponseField name="totalWin" type="number">
      Total number of winning trades
    </ResponseField>
    <ResponseField name="totalLose" type="number">
      Total number of losing trades
    </ResponseField>
    <ResponseField name="totalLiquidation" type="number">
      Total number of liquidations
    </ResponseField>
    <ResponseField name="totalLiquidationAmount" type="number">
      Total amount liquidated
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Timestamp when the entry was created
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Metadata about the response
  <Expandable title="properties">
    <ResponseField name="limit" type="number">
      Number of records returned
    </ResponseField>
    <ResponseField name="offset" type="number">
      Number of records skipped
    </ResponseField>
    <ResponseField name="total" type="number">
      Total number of records matching the query
    </ResponseField>
    <ResponseField name="totalPages" type="number">
      Total number of pages
    </ResponseField>
  </Expandable>
</ResponseField>
