---
title: "Order List"
api: "POST https://api.copin.io/{protocol}/order/filter"
description: "Retrieve orders for a specific protocol"
---
## Authentication
This endpoint requires authentication using a JWT token in the Authorization header.

<ParamField header="Authorization" type="string" required>
    JWT token in the format: `Authorization {token}`
</ParamField>

<ParamField header="x-api-key" type="string">
    Token in the format: `x-api-key {api-key}`
</ParamField>

## Path Parameters

<ParamField path="protocol" type="string" required>
  The protocol to get positions for. Supported values include: `GMX`, `KWENTA`,
  `POLYNOMIAL`, `POLYNOMIAL_L2`, `GMX_V2_AVAX`, `GMX_V2`, `GNS`, `GNS_POLY`,
  `GNS_BASE`, `GNS_APE`, `LEVEL_BNB`, `LEVEL_ARB`, `MUX_ARB`, `APOLLOX_BNB`,
  `AVANTIS_BASE`, `EQUATION_ARB`, `LOGX_BLAST`, `LOGX_MODE`, `MYX_ARB`,
  `DEXTORO`, `VELA_ARB`, `HMX_ARB`, `SYNTHETIX_V3`, `SYNTHETIX_V3_ARB`,
  `KTX_MANTLE`, `CYBERDEX`, `YFX_ARB`, `KILOEX_OPBNB`, `KILOEX_BNB`,
  `KILOEX_MANTA`, `KILOEX_BASE`, `ROLLIE_SCROLL`, `MUMMY_FANTOM`, `HYPERLIQUID`,
  `SYNFUTURE_BASE`, `MORPHEX_FANTOM`, `PERENNIAL_ARB`, `BSX_BASE`, `DYDX`,
  `UNIDEX_ARB`, `VERTEX_ARB`, `HORIZON_BNB`, `HOLDSTATION_ZKSYNC`, `ZENO_METIS`,
  `LINEHUB_LINEA`, `BMX_BASE`, `FOXIFY_ARB`, `APOLLOX_BASE`, `GMX_AVAX`,
  `SYNTHETIX`, `DEPERP_BASE`, `ELFI_ARB`
</ParamField>

## Request Body

<ParamField body="pagination" type="object">
  Pagination parameters
  <Expandable title="properties">
    <ParamField body="limit" type="number" default="20">
      Number of records to return
    </ParamField>
    <ParamField body="offset" type="number" default="0">
      Number of records to skip
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="sortBy" type="string">
  Field to sort by. Values: `blockTime`, `blockNumber`.
</ParamField>

<ParamField body="sortType" type="string" default="desc">
  Sort direction. Values: `desc`, `asc`
</ParamField>

<ParamField body="queries" type="array">
  Array of query objects
  <Expandable title="properties">
    <ParamField body="fieldName" type="string">
      Field name to query. Values: `account`, `status`, `pair`
    </ParamField>
    <ParamField body="value" type="string">
      Value for the field. For type: `OPEN`, `INCREASE`, `DECREASE`, 'CLOSE'. For account: address
      string. For pair: `ETH-USDT`, `BTC-USDT`,...
    </ParamField>
  </Expandable>
</ParamField>

## Request Example

```json
{
  "pagination": {
    "limit": 20,
    "offset": 0
  },
  "queries": [
    {
      "fieldName": "account",
      "value": "0xA142bF2CAc3B60157a31741BA45D33e84c3D1f39"
    }
  ],
  "sortBy": "blockTime",
  "sortType": "desc"
}
```
