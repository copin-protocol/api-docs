---
title: "Copy Positions"
api: "POST https://api.copin.io/copy-positions/page"
description: "Retrieve copy positions with pagination"
---

## Authentication
This endpoint requires authentication using a JWT token in the Authorization header.

<ParamField header="Authorization" type="string" required>
    JWT token in the format: `Authorization {token}`
</ParamField>

## Request

<ParamField query="limit" type="number" default="20">
    Number of records to return per page
</ParamField>
<ParamField query="offset" type="number" default="0">
    Number of records to skip for pagination
</ParamField>
<ParamField query="status" type="string">
    Current status of the position ("OPEN/CLOSE")
</ParamField>
<ParamField query="isLong" type="boolean">
    Position side
</ParamField>

## Response
The response includes a data array containing position objects and a meta object with pagination information.

<ResponseField name="data" type="array">
    Array of position objects
</ResponseField>
<ResponseField name="data[].id" type="string">
    Unique identifier for the copy position
</ResponseField>
<ResponseField name="data[].userId" type="string">
    Identifier of the user who owns this position
</ResponseField>
<ResponseField name="data[].copyTradeId" type="string">
    Identifier of the copy trade configuration that created this position
</ResponseField>
<ResponseField name="data[].copyAccount" type="string">
    The wallet address being copied from
</ResponseField>
<ResponseField name="data[].positionIndex" type="number">
    Index of the position in the original account
</ResponseField>
<ResponseField name="data[].copyPositionIndex" type="number">
    Index of the position in the copy account
</ResponseField>
<ResponseField name="data[].indexToken" type="string">
    Token identifier for the indexed token
</ResponseField>
<ResponseField name="data[].identifyKey" type="string">
    Unique key to identify the position
</ResponseField>
<ResponseField name="data[].pair" type="string">
    Trading pair (e.g., "ETH-USDT")
</ResponseField>
<ResponseField name="data[].key" type="string">
    Unique key for the position
</ResponseField>
<ResponseField name="data[].sourceOrderTxHashes" type="array">
    Array of transaction hashes for the source orders
</ResponseField>
<ResponseField name="data[].sourceSizeDelta" type="string">
    Size delta in the source position
</ResponseField>
<ResponseField name="data[].sizeDelta" type="string">
    Size delta in the copy position
</ResponseField>
<ResponseField name="data[].totalSizeDelta" type="number">
    Total size delta for the position
</ResponseField>
<ResponseField name="data[].sourceInitialVol" type="string">
    Initial volume in the source position
</ResponseField>
<ResponseField name="data[].initialVol" type="number">
    Initial volume in the copy position
</ResponseField>
<ResponseField name="data[].isLong" type="boolean">
    Whether the position is long (true) or short (false)
</ResponseField>
<ResponseField name="data[].entryPrice" type="number">
    Entry price for the position
</ResponseField>
<ResponseField name="data[].leverage" type="number">
    Leverage used for the position
</ResponseField>
<ResponseField name="data[].stopLossPrice" type="number">
    Price at which stop loss is triggered
</ResponseField>
<ResponseField name="data[].stopLossAmount" type="number">
    Amount for stop loss
</ResponseField>
<ResponseField name="data[].takeProfitPrice" type="number">
    Price at which take profit is triggered
</ResponseField>
<ResponseField name="data[].takeProfitAmount" type="number">
    Amount for take profit
</ResponseField>
<ResponseField name="data[].pnl" type="number">
    Total profit and loss
</ResponseField>
<ResponseField name="data[].realisedPnl" type="number">
    Realised profit and loss
</ResponseField>
<ResponseField name="data[].funding" type="number">
    Funding payments for the position
</ResponseField>
<ResponseField name="data[].fee" type="number">
    Fees paid for the position
</ResponseField>
<ResponseField name="data[].closeType" type="string">
    How the position was closed (COPY_TRADE/MANUAL/FORCE_CLOSE/TAKE_PROFIT/STOP_LOSS/LIQUIDATE/OVERWRITE)
</ResponseField>
<ResponseField name="data[].status" type="string">
    Current status of the position (OPEN/CLOSE)
</ResponseField>
<ResponseField name="data[].protocol" type="string">
    Protocol used for the position
</ResponseField>
<ResponseField name="data[].exchange" type="string">
    Exchange where the position was opened
</ResponseField>
<ResponseField name="data[].isReverse" type="boolean">
    Whether this is a reverse copy position
</ResponseField>
<ResponseField name="data[].lastOrderAt" type="string">
    Timestamp of the last order for this position
</ResponseField>
<ResponseField name="data[].createdAt" type="string">
    Timestamp when the position was created
</ResponseField>
<ResponseField name="data[].copyTradeTitle" type="string">
    Title of the copy trade configuration
</ResponseField>
<ResponseField name="meta" type="object">
    Pagination metadata
</ResponseField>
<ResponseField name="meta.limit" type="number">
    Number of records per page
</ResponseField>
<ResponseField name="meta.offset" type="number">
    Number of records skipped
</ResponseField>
<ResponseField name="meta.total" type="number">
    Total number of records available
</ResponseField>
<ResponseField name="meta.totalPages" type="number">
    Total number of pages available
</ResponseField>
